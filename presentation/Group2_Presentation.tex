\section{Interpretation}
% ============================================================
\begin{frame}{Interpretation --- Generalised Gamma Model}

\begin{columns}[T]

  \begin{column}{0.46\textwidth}
    \vspace{4pt}
    \begin{table}
      \caption{MLE Parameter Estimates with 95\% Confidence Intervals}
      \small
      \begin{tabular}{@{}lrrl@{}}
        \toprule
        \textbf{Parameter} & \textbf{Estimate} & \textbf{SE} & \textbf{95\% CI} \\
        \midrule
        $\mu$      & 7.545    & 0.117 & {[}7.317,\ 7.774{]} \\
        $\sigma$   & 1.575    & 0.073 & {[}1.438,\ 1.725{]} \\
        $Q$        & $-0.481$ & 0.191 & {[}$-0.855$,\ $-0.106${]} \\
        \bottomrule
      \end{tabular}
    \end{table}

    \vspace{8pt}
    \small
    \begin{itemize}
      \item \textbf{$\mu = 7.545$:} median survival
            $\approx e^{7.545} \approx 1{,}900$ days
      \item \textbf{$\sigma = 1.575$:} spread of the distribution
      \item \textbf{$Q = -0.481 < 0$:} hazard
            \alert{rises then falls} --- non-monotone
    \end{itemize}
  \end{column}

  \begin{column}{0.50\textwidth}
    \vspace{14pt}
    \small
    \textbf{Life function interpretation:}
    \begin{itemize}
      \item \textbf{S(t):} 90\% survive at 1 year;
            $\approx$55\% at 5 years
      \item \textbf{h(t):} Risk peaks near day $\sim$300 then
            \alert{decreases} --- consistent with post-surgical
            recovery followed by stabilisation
      \item \textbf{f(t):} Most deaths occur around 300 days
            after diagnosis
      \item \textbf{F(t):} 50\% of patients die within
            $\approx$2{,}500 days ($\sim$6.8 years)
    \end{itemize}

    \vspace{8pt}
    \textbf{Mean survival:} 2{,}130 days \textbf{(5.84 years)}\\
    High SD (1{,}221 days) reflects substantial heterogeneity.
  \end{column}

\end{columns}
\end{frame}

\section{Conclusions \& Insights}
% ============================================================
\begin{frame}{Conclusions \& Clinical Insights}

\begin{columns}[T]

  \begin{column}{0.48\textwidth}
    \textbf{Statistical Conclusions}
    \vspace{4pt}
    \begin{itemize}
      \small
      \item \textbf{Generalised Gamma} best fits colon cancer
            survival data (AIC = 8{,}212.9)
      \item It allows a flexible hazard shape, outperforming
            simpler Weibull and Exponential models
      \item $Q = -0.481 < 0$: hazard is non-monotone ---
            rises early then gradually declines
      \item High variability (SD = 1{,}221 days) suggests
            important patient-level heterogeneity
    \end{itemize}
  \end{column}

  \begin{column}{0.48\textwidth}
    \textbf{Clinical Implications}
    \vspace{4pt}
    \begin{itemize}
      \small
      \item \textbf{High early risk:} The first $\sim$300 days
            carry the highest death hazard --- intensive
            post-operative monitoring is critical
      \item \textbf{Mean survival $\approx$ 5.8 years:}
            Provides a practical benchmark for care planning
      \item $\sim$45\% of patients survive beyond 3{,}000 days,
            indicating a favourable long-term subgroup
      \item Lymph node count (\texttt{nodes}) should be
            incorporated in extended covariate models
    \end{itemize}
  \end{column}

\end{columns}

\vspace{10pt}
\begin{center}
  \small
  \textit{The non-monotone hazard pattern is clinically meaningful:
  early intensive follow-up and long-term survivorship planning
  are both warranted for colon cancer patients.}
\end{center}

\end{frame}
